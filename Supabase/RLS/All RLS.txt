-- 1. Проверка кода восстановления (SECURITY DEFINER, доступна анонимно)
CREATE OR REPLACE FUNCTION verify_recovery_code(p_user_id uuid, p_code text)
RETURNS boolean
LANGUAGE sql
SECURITY DEFINER
AS $$
  SELECT EXISTS (
    SELECT 1 FROM recovery_codes
    WHERE user_id = p_user_id AND code_hash = p_code AND used = false
  );
$$;

-- 2. Получение Telegram‑настроек по user_id (доступна анонимно)
CREATE OR REPLACE FUNCTION get_telegram_settings(p_user_id uuid)
RETURNS TABLE(telegram_token text, telegram_chat_id text)
LANGUAGE sql
SECURITY DEFINER
AS $$
  SELECT telegram_token, telegram_chat_id
  FROM settings
  WHERE user_id = p_user_id
  LIMIT 1;
$$;

-- 3. Обновлённая функция поиска пользователя (SECURITY DEFINER)
DROP FUNCTION IF EXISTS get_user_by_username;
CREATE OR REPLACE FUNCTION get_user_by_username(p_username text)
RETURNS TABLE(id uuid, email text)
LANGUAGE sql
SECURITY DEFINER
AS $$
  SELECT id, email::text
  FROM auth.users
  WHERE raw_user_meta_data ->> 'username' = p_username
  LIMIT 1;
$$;

  USING (auth.uid() = user_id);

-- >>> settings (особая логика: управляет только владелец, видят все участники) <<<
DROP POLICY IF EXISTS "Access own or shared car settings" ON settings;
CREATE POLICY "Select settings for shared car" ON settings FOR SELECT
  USING (can_access_car(car_id));
CREATE POLICY "Owner can insert settings" ON settings FOR INSERT
  WITH CHECK (
    auth.uid() = user_id
    AND EXISTS (SELECT 1 FROM cars WHERE id = car_id AND user_id = auth.uid())
  );
CREATE POLICY "Owner can update settings" ON settings FOR UPDATE
  USING (
    auth.uid() = user_id
    AND EXISTS (SELECT 1 FROM cars WHERE id = car_id AND user_id = auth.uid())
  )
  WITH CHECK (
    auth.uid() = user_id
    AND EXISTS (SELECT 1 FROM cars WHERE id = car_id AND user_id = auth.uid())
  );
CREATE POLICY "Owner can delete settings" ON settings FOR DELETE
  USING (
    auth.uid() = user_id
    AND EXISTS (SELECT 1 FROM cars WHERE id = car_id AND user_id = auth.uid())
  );
